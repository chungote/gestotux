00103         cursor->insertText( QString( "Establecimiento:  %1\n" ).arg( colaAuxiliar->record().value(0).toString() ) );

00105  else
00106  { qDebug( "Error al ejecutar la cola de nombre de establecimiento" ); }
00107  // Busco las caravanas que estan en ese establecimiento
00108  colaAuxiliar->exec( QString( "SELECT codigo FROM car_caravana WHERE id_caravana IN (  SELECT id_caravana FROM car_carv_tri WHERE id_tri IN (  SELECT id_tri FROM car_tri WHERE ( id_estab_origen = '%1' OR id_estab_destino = '%1' ) AND razon IN ( 2, 3 ) ) )" ).arg( _idEstablecimiento ) );
00109  if( colaAuxiliar->size() == 0 )
00110  {
00111    cursor->movePosition( QTextCursor::End );
00112    cursor->insertText( "\n\nNo existen resultados" );
00113  }
00114  else
00115  {
00116         // Genero la tabla
00117         cursor->movePosition( QTextCursor::End );
00118         tabla = cursor->insertTable( 1, 2 );
00119         QTextTableFormat formatoTabla = tabla->format();
00120         formatoTabla.setHeaderRowCount(1);
00121         tabla->setFormat( formatoTabla );
00122         tabla->cellAt( 0,0 ).firstCursorPosition().insertHtml( "#Num" );
00123         tabla->cellAt( 0,1 ).firstCursorPosition().insertHtml( "#Caravana" );
00124         while( colaAuxiliar->next() )
00125         {
00126                 int pos = tabla->rows();

00128                 tabla->cellAt( pos, 0 ).firstCursorPosition().insertHtml( QString( "%L1" ).arg( pos ) );
00129                 tabla->cellAt( pos, 1 ).firstCursorPosition().insertHtml( colaAuxiliar->record().value(0).toString() );
00130         }
00131  } // fin else cola != 0
00132  delete colaAuxiliar;
