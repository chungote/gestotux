00045 void EEnvioBackup::run()
00046 {
00047  preferencias *p = preferencias::getInstancia();
00048  if( p->value( "backup/enviado", false ).toBool() == true  )
00049  {
00050   qDebug( "El backup ya ha sido enviado... saliendo del hilo" );
00051   exit(0);
00052   return;
00053  }
00054  setTerminationEnabled();
00055  archivo = new QFile( p->value( "backup/archivo", QDate::currentDate().toString() ).toString() );
00056  if( !archivo->open( QIODevice::ReadOnly ) )
00057  {
00058   qDebug( QString( "Error al abrir el archivo de backup: %1" ).arg( p->value( "backup/archivo" ).toString() ).toLocal8Bit() );
00059   delete archivo;
00060   exit(0);
00061   return;
00062  }
00063  ftp = new QFtp();
00064  connect( ftp, SIGNAL( commandFinished( int, bool ) ), this, SLOT( finComando( int, bool ) ) );
00065  ftp->connectToHost( p->value( "ftp/host", "tranfuga.no-ip.org" ).toString(), p->value( "ftp/puerto", 21 ).toInt() );
00066  ftp->login();
00067  ftp->cd( gestotux::pluginInfo()->directorioBackup() );
00068  ftp->put( archivo, QDateTime::currentDateTime().toString( "yyyyMMddhhmmsszzz") );
00069  ftp->close();
00070  exec();
00071 }
